### 方式 3: 自建监控脚本

**文件**: `skills/shanmu/xiaohongshu_monitor.py`

```python
#!/usr/bin/env python3
# 小红书数据监控脚本

class XiaohongshuMonitor:
    def __init__(self):
        self.account_id = "sayelf_tea"
        self.data_file = "/data/xiaohongshu_data.json"
    
    def get_account_stats(self):
        """获取账号数据"""
        return {
            "followers": 0,  # 粉丝数
            "notes": 0,  # 笔记数
            "likes": 0,  # 总点赞
            "collects": 0,  # 总收藏
            "comments": 0  # 总评论
        }
    
    def get_note_performance(self, note_id):
        """获取笔记表现"""
        return {
            "views": 0,  # 阅读量
            "likes": 0,  # 点赞
            "collects": 0,  # 收藏
            "comments": 0  # 评论
        }
    
    def daily_report(self):
        """生成日报"""
        data = self.get_account_stats()
        report = f"""
【小红书日报 · {datetime.now().strftime('%Y-%m-%d')}】

粉丝数：{data['followers']}
笔记数：{data['notes']}
总点赞：{data['likes']}
总收藏：{data['collects']}
总评论：{data['comments']}

今日新增：
- 粉丝：+{data['new_followers']}
- 点赞：+{data['new_likes']}
- 收藏：+{data['new_collects']}
"""
        return report
```
