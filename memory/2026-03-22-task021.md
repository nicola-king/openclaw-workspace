```bash
import ezdxf

# 读取 DXF 文件
doc = ezdxf.readfile("test.dxf")
msp = doc.modelspace()

# 提取几何信息
for entity in msp:
    print(f"类型：{entity.dxftype()}")
    if entity.dxftype() == 'LINE':
        print(f"  起点：{entity.dxf.start}")
        print(f"  终点：{entity.dxf.end}")
    elif entity.dxftype() == 'CIRCLE':
        print(f"  圆心：{entity.dxf.center}")
        print(f"  半径：{entity.dxf.radius}")
EOF
```
