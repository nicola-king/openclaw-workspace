## 🚀 实现代码 (小程序)

```javascript
// components/StageCard/index.js
Component({
  properties: {
    stage: Object,
    step: Number,
    type: String
  },
  
  methods: {
    onSaveImage() {
      // 生成卡片图片
      wx.canvasToTempFilePath({
        canvasId: 'stage-card',
        success: (res) => {
          wx.saveImageToPhotosAlbum({
            filePath: res.tempFilePath
          })
        }
      })
    },
    
    onShare() {
      // 分享朋友圈
      wx.shareAppMessage({
        title: this.data.stage.name,
        imageUrl: this.data.cardImage
      })
    }
  }
})
```
